/*
	This query is intended to be run in 'DAX Studio.
	It is designed to speed up report documentation.
	Change the 'Results' dropdown to 'Static' in the 'Home' tab > 'Output' option.
*/

DEFINE

-- REFERENCE

VAR _refIteratorFunctions = 
	{
		"SUMX(",
		"AVERAGEX(",
		"MINX(",
		"MAXX(",
		"COUNTX(",
		"RANKX(",
		"PRODUCTX(",
		"FILTER(",
		"CALCULATETABLE(",
		"ADDCOLUMNS(",
		"GENERATE(",
		"GENERATEALL(",
		"GROUPBY(",
		"SELECTCOLUMNS(",
		"SUMMARIZE(",
		"SUMMARIZECOLUMNS("
	}
VAR _refRefreshGranularity =
	DATATABLE("Granularity",STRING,"Granularity ID",INTEGER,
		{
			{"Invalid", -1},
			{"Day(s)", 0},
			{"Month(s)", 1},
			{"Quarter(s)", 2},
			{"Year(s)", 3}
		}
	)
VAR _refRefreshMode =
	DATATABLE("Refresh Mode", STRING, "Refresh Mode ID", INTEGER,
		{
			{"Import", 0},
			{"Hybrid", 1}
		}
	)
VAR _refRlsRoles =
	SELECTCOLUMNS(
		INFO.ROLES(),
		"RLS Role ID", [ID],
		"Role", [Name],
		"Description", [Description]
	)
VAR _refTables =
	SELECTCOLUMNS(
		INFO.TABLES(),
			"Table ID", [ID],
			"Table", [Name]
	)
	
-- BASE

VAR _baseColumns =
	NATURALLEFTOUTERJOIN(
		SELECTCOLUMNS(
			INFO.VIEW.COLUMNS(),
			"Columns", BLANK(),
			[Table],
			"Column", [Name],
			[Description],
			"DAX", [Expression],
			"Format String", [FormatString],
			"Data Type", [DataType],
			"Is Hidden", [IsHidden],
			"Is Unique", [IsUnique],
			"Sorted By", [SortByColumn],
			"Category", [DataCategory]
		),
		SELECTCOLUMNS(
			COLUMNSTATISTICS(),
			"Table", [Table Name],
			"Column", [Column Name],
			"Lowest Value", CONVERT([Min], STRING),
			"Highest Value", CONVERT([Max], STRING),
			[Cardinality],
			"Value Length", [Max Length]
		)
	)
VAR _basePartitions =
	VAR _partitionRows =
		SELECTCOLUMNS(
			NATURALLEFTOUTERJOIN(
				NATURALLEFTOUTERJOIN(
					SELECTCOLUMNS(
						INFO.PARTITIONS(),
						"Partition ID", [ID],
						"PartitionStorageID", [PartitionStorageID]
					),
					SELECTCOLUMNS(
						INFO.PARTITIONSTORAGES(),
						"PartitionStorageID", [ID],
						"SegmentMapStorageID", [SegmentMapStorageID]
					)
				),
				SELECTCOLUMNS(
					INFO.SEGMENTMAPSTORAGES(),
					"SegmentMapStorageID", [ID],
					"Rows", [RecordCount]
				)
			),
			[Partition ID],
			[Rows]
		)
	RETURN
		NATURALLEFTOUTERJOIN(
			SELECTCOLUMNS(
				INFO.PARTITIONS(),
				"Partition ID", [ID],
				"Table ID", [TableID],
				"Table", MINX(FILTER(_refTables, [Table ID] = [TableID]), [Table]),
				"Partition", [Name],
				"Refresh Datetime",
					IF(
						[RefreshedTime] < DATE(1900, 1, 1),
						COALESCE([ModifiedTime], [RefreshedTime]),
						[RefreshedTime]
					),
				"Partition Query", [QueryDefinition],
				"RangeStart", [RangeStart],
				"RangeEnd", [RangeEnd]
			),
			_partitionRows
		)
VAR _baseTables =
	VAR _baseTableRows =
		SELECTCOLUMNS(
			FILTER(_baseColumns, [Category] = "RowNumber"),
			[Table],
			"Rows", [Cardinality]
		)
	VAR _baseTablePartitions =
		SELECTCOLUMNS(
			GROUPBY(
				_basePartitions,
				[Table ID],
				"Refresh Datetime", MAXX(CURRENTGROUP(), [Refresh Datetime]),
				"Partition Query", MAXX(CURRENTGROUP(), [Partition Query])
			),
			[Table ID], [Refresh Datetime], [Partition Query]
		)
	VAR _baseTableQueries =
		VAR _baseRefreshPolicyQueries =
			SELECTCOLUMNS(
				INFO.REFRESHPOLICIES(),
				"Table ID", [TableID],
				"Refresh Policy Query", [SourceExpression]
			)
		RETURN
			SELECTCOLUMNS(
				NATURALLEFTOUTERJOIN(
					_baseTablePartitions,
					_baseRefreshPolicyQueries
				),
				"DAX or M-Query", COALESCE([Refresh Policy Query], [Partition Query]),
				[Table ID], [Refresh Datetime], [Partition Query]
			)
	RETURN
		NATURALLEFTOUTERJOIN(
			NATURALLEFTOUTERJOIN(
				SELECTCOLUMNS(
                    ADDCOLUMNS(
					    INFO.VIEW.TABLES(),
					"Is Calculated", IF([Expression] <> BLANK(), TRUE, FALSE)
                    ),
					"Tables", BLANK(),
					"Table ID", [ID],
					"Table", [Name],
					[Description],
					"Is Hidden", [IsHidden],
					"Storage Mode", [StorageMode],
                    [Is Calculated]
				),
				_baseTableRows
			),
			_baseTableQueries
		)
VAR _baseDependencies =
	SELECTCOLUMNS(
		INFO.CALCDEPENDENCY(),
		"Object", [OBJECT],
		"Table", [TABLE],
		"Type", [OBJECT_TYPE],
		"Referenced Object", [REFERENCED_OBJECT],
		"Referenced Table", [REFERENCED_TABLE],
		"Referenced Type", [REFERENCED_OBJECT_TYPE]
	)
VAR _baseCalculationGroups =
	SELECTCOLUMNS(
		INFO.CALCULATIONGROUPS(),
		"Calculation Group ID", [ID],
		"Table ID", [TableID],
		"Table", MINX(FILTER(_refTables, [Table ID] = [TableID]), [Table]),
		[Description]
	)

-- FINAL

VAR _tables =
	SELECTCOLUMNS(
		_baseTables,
		[Tables], [Table], [Description], 
		"Refresh Datetime UTC", [Refresh Datetime],
		[Is Hidden], [Is Calculated], [Storage Mode], [Rows], [DAX or M-Query]
	)
VAR _columns =
	SELECTCOLUMNS(
		FILTER(
			_baseColumns,
			[Category] <> "RowNumber"
		),
		[Columns], [Table], [Column],
		[Description],[DAX],[Format String], [Data Type], [Is Hidden], [Is Unique], [Sorted By], [Lowest Value], [Highest Value], [Cardinality],
		"Value Length", COALESCE([Value Length], LEN([Highest Value]))
	)
VAR _measures =
	NATURALLEFTOUTERJOIN(
		NATURALLEFTOUTERJOIN(
			SELECTCOLUMNS(
				INFO.VIEW.MEASURES(),
				"Measures", BLANK(),
				"Measure", [Name],
				[Description],
				"Data Type", [DataType],
				"DAX", [Expression], 
				"Format String", [FormatStringDefinition], 
				[Table], 
				"Display Folder", [DisplayFolder],
				"Is Hidden", [IsHidden],
				[State],
				"DAX Length", LEN([Expression]),
				"Is Iterator",
					IF (
						COUNTROWS(
							FILTER(
								_refIteratorFunctions,
								CONTAINSSTRING(
									SUBSTITUTE(
									SUBSTITUTE(
									SUBSTITUTE(
									SUBSTITUTE(UPPER([Expression]),
										" ",""),
										UNICHAR(9), ""),
										UNICHAR(10), ""),
										UNICHAR(13), ""),
									[Value]
								)
							)
						) > 0,
						TRUE,
						FALSE
					),
				"Variables",
					VAR _exp =
						SUBSTITUTE(
						SUBSTITUTE(
						SUBSTITUTE(
						SUBSTITUTE(UPPER([Expression]),
							" ",""),
							"|", "~"),
							UNICHAR(9), ""),
							UNICHAR(10) & "VAR","|")
					VAR _cnt = PATHLENGTH(_exp) - 1 + IF(LEFT(_exp, 3) = "VAR", 1,0)
					RETURN
						IF(_cnt <= 0,BLANK(), _cnt)
			),
			SELECTCOLUMNS(
				GROUPBY(FILTER(_baseDependencies, [Referenced Type] = "MEASURE" && [Type] = "MEASURE"), [Object], "Referenced Measures", COUNTX(CURRENTGROUP(),1)),
					"Measure", [Object],
					[Referenced Measures]
			)
		),
		SELECTCOLUMNS(
		GROUPBY(FILTER(_baseDependencies, [Type] = "MEASURE"), [Object], "Referenced Objects", COUNTX(CURRENTGROUP(),1)),
			"Measure", [Object],
			[Referenced Objects]
		)
	)
VAR _relationships =
	SELECTCOLUMNS(
		INFO.VIEW.RELATIONSHIPS(),
		"Relationships", BLANK(),
		[Relationship],
		[Name],
		"Is Active", [IsActive],
		"Direction", [CrossFilteringBehavior],
		"From Table", [FromTable],
		"From Column", [FromColumn],
		"To Table", [ToTable],
		"To Column", [ToColumn],
		"Cardinality", [FromCardinality] & " to " & [ToCardinality]
	)
VAR _otherQueries =
	SELECTCOLUMNS(
		INFO.EXPRESSIONS(),
		"Other Queries", BLANK(),
		"Query Name", [Name],
		"Description", [Description],
		"M-Query", [Expression]
	)
VAR _incrementalTables =
	SELECTCOLUMNS(
		INFO.REFRESHPOLICIES(),
		"Incremental Tables", BLANK(),
		"Table", MINX(FILTER(_refTables, [Table ID] = [TableID]), [Table]),
		"Rolling Window History", [RollingWindowPeriods] & " " & MINX(FILTER(_refRefreshGranularity, [RollingWindowGranularity] = [Granularity ID]), [Granularity]),
		"Incremental Window", [IncrementalPeriods] & " " & MINX(FILTER(_refRefreshGranularity, [IncrementalGranularity] = [Granularity ID]), [Granularity]),
		"Incremental Window End",  [IncrementalPeriodsOffset] & " " & MINX(FILTER(_refRefreshGranularity, [IncrementalGranularity] = [Granularity ID]), [Granularity]) & " from Refresh Date",
		"Refresh Mode", MINX(FILTER(_refRefreshMode, [Refresh Mode ID] = [PolicyType]), [Refresh Mode])
	)
VAR _incrementalPartitions =
	VAR _baseIncrementalTables =
		SELECTCOLUMNS(
			FILTER(
				GROUPBY(
					_basePartitions,
					[Table ID],
					"Partition Count", COUNTX(CURRENTGROUP(), 1)
				),
				[Partition Count] > 1
			),
			[Table ID]
		)
	RETURN
		SELECTCOLUMNS(
			NATURALINNERJOIN(
				_basePartitions,
				_baseIncrementalTables
			),
			"Incremental Partitions", BLANK(),
			[Table], [Partition], [Rows],
			"Refresh/Merge Datetime UTC", [Refresh Datetime],
			[RangeStart], [RangeEnd]
		)
VAR _rlsRoles =
	SELECTCOLUMNS(_refRlsRoles,
		"RLS Roles", BLANK(),
		[Role],
		[Description]
	)
VAR _rlsFilters =
	VAR _baseRlsFilters =
		SELECTCOLUMNS(
			INFO.TABLEPERMISSIONS(),
			"Role ID", [RoleID],
			"RLS Table ID", [TableID],
			"DAX", [FilterExpression]
		)
	RETURN
		SELECTCOLUMNS(
			_baseRlsFilters,
			"RLS Filters", BLANK(),
			"Role", MINX(FILTER(_refRlsRoles, [Role ID] = [RLS Role ID]), [Role]),
			"Table", MINX(FILTER(_refTables, [Table ID] = [RLS Table ID]), [Table]),
			[DAX]
		)
VAR _calcGroups =
	SELECTCOLUMNS(
		_baseCalculationGroups,
		"Calculation Groups", BLANK(),
		"Calculation Group", [Table],
		[Description]
	)
VAR _calcItems =
	SELECTCOLUMNS(
		INFO.CALCULATIONITEMS(),
		"Calculation Items", BLANK(),
		"Calculation Group", MINX(FILTER(_baseCalculationGroups, [Calculation Group ID] = [CalculationGroupID]), [Table]),
		"Calculation Item", [Name],
		[Description],
		"DAX", [Expression]
	)
VAR _dependencies =
	SELECTCOLUMNS(
		_baseDependencies,
		"Dependencies", BLANK(),
		[Object], [Table], [Type], [Referenced Object], [Referenced Table], [Referenced Type]
	)
VAR _info =
	{
        (1,"POWER BI DOCUMENTATION","Script: ","https://github.com/fulopattilak/data-and-bi/wiki/Report-documentation-with-DAX-Studio"),
		(2,"Created at (UTC): ",UTCNOW(),),
		(3,"Report refresh (UTC): ",MINX(_tables, [Refresh Datetime UTC]),),
        (4,,,),
		(5,"Sheet", "Table Name", "Description"),
		(6,,,),
		(7,"Query 2", "Tables", "Describes the tables loaded from Power Query and all calculated tables."),
		(8,"Query 3", "Columns", "Describes columns and calculated columns."),
		(9,"Query 4", "Measures", "Describes all measures."),
		(10,"Query 5", "Relationships","Describes the relationships between tables."),
		(11,"Query 6", "Other Queries","Describes the queries that are used in Power Query, but not loaded to the model."),
		(12,"Query 7", "Incremental Tables", "Describes the incremental refresh policy of the tables."),
		(13,"Query 8", "Incremental Partitions", "Lists partitions for tables that have more than one partition."),
		(14,"Query 9", "RLS Roles", "Lists the RLS roles that are set."),
		(15,"Query 10", "RLS Filters", "Describes the filtering criteria for RLS roles."),
		(16,"Query 11", "Calculation Groups", "Lists Calculation groups."),
		(17,"Query 12", "Calculation Items", "Describes the Calculation items defined inside Calculation groups."),
		(18,"Query 13", "Dependencies", "Describes table column and measure dependencies.")
	}
		--(,"Query 14", "User Functions", "Lists User Defined Functions.")
/*
	VAR _functions =
		SELECTCOLUMNS(
			INFO.USERDEFINEDFUNCTIONS(),
			"User Functions", BLANK(),
			[Name],
			[Description],
			"DAX", [Expression],
			"Is Hidden", [IsHidden]
		)
*/

-- OUTPUT

EVALUATE _info ORDER BY [Value1]
EVALUATE _tables ORDER BY [Table]
EVALUATE _columns ORDER BY [Table], [Column]
EVALUATE _measures ORDER BY [Measure]
EVALUATE _relationships ORDER BY [Relationship]
EVALUATE _otherQueries ORDER BY [Query Name]
EVALUATE _incrementalTables ORDER BY [Table]
EVALUATE _incrementalPartitions ORDER BY [Table], [Partition]
EVALUATE _rlsRoles ORDER BY [Role]
EVALUATE _rlsFilters ORDER BY [Role], [Table]
EVALUATE _calcGroups ORDER BY [Calculation Group]
EVALUATE _calcItems ORDER BY [Calculation Group], [Calculation Item]
EVALUATE _dependencies ORDER BY [Table], [Object], [Referenced Table], [Referenced Object]
-- EVALUATE _functions
