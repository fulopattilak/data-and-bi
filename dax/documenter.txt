/*
This query is intended to be run in 'DAX Studio.
It is designed to speed up report documentation.
Change the 'Results' dropdown to 'Static' in the 'Home' tab > 'Output' option.
*/

DEFINE
 
VAR _info = 
	{
		("POWER BI",,"Created at: ",FORMAT(UTCNOW(), "d. mmm yyyy, h:nn") & " UTC"),
		("DOCUMENTATION",,"Script: ","https://github.com/fulopattilak/data-and-bi/wiki/Report-documentation-with-DAX-Studio"),
		(,,,),
		(,"Sheet", "Table Name", "Description"),
		(,,,),
		(,"Query 2", "Tables", "Describes the tables loaded from Power Query and all calculated tables."),
		(,"Query 3", "Columns", "Describes columns and calculated columns."),
		(,"Query 4", "Measures", "Describes all measures."),
		(,"Query 5", "Relationships","Describes the relationships between tables."),
		(,"Query 6", "Other Queries","Describes the queries that are used in Power Query, but not loaded to the model."),
		(,"Query 7", "Incremental Tables", "Describes the incremental refresh policy of the tables."),
		(,"Query 8", "RLS Roles", "Lists the RLS roles that are set."),
		(,"Query 9", "RLS Filters", "Describes the filtering criteria for RLS roles."),
		(,"Query 10", "Calculation Groups", "Lists Calculation groups."),
		(,"Query 11", "Calculation Items", "Describes the Calculation items defined inside Calculation groups."),
		(,"Query 12", "Dependencies", "Describes table column and measure dependencies.")
	}
		--(,"Query 13", "User Functions", "Lists User Defined Functions.")

-- REFERENCE

VAR _refIteratorFunctions = 
	{
		"SUMX(",
		"AVERAGEX(",
		"MINX(",
		"MAXX(",
		"COUNTX(",
		"RANKX(",
		"PRODUCTX(",
		"FILTER(",
		"CALCULATETABLE(",
		"ADDCOLUMNS(",
		"GENERATE(",
		"GENERATEALL(",
		"GROUPBY(",
		"SELECTCOLUMNS(",
		"SUMMARIZE(",
		"SUMMARIZECOLUMNS("
	}
VAR _refRefreshGranularity =
	DATATABLE("Granularity",STRING,"Granularity ID",INTEGER,
		{
			{"Invalid", -1},
			{"Day(s)", 0},
			{"Month(s)", 1},
			{"Quarter(s)", 2},
			{"Year(s)", 3}
		}
	)
VAR _refRefreshMode =
	DATATABLE("Refresh Mode", STRING, "Refresh Mode ID", INTEGER,
		{
			{"Import", 0},
			{"Hybrid", 1}
		}
	)
VAR _refRlsRoles =
	SELECTCOLUMNS(
		INFO.ROLES(),
		"RLS Role ID", [ID],
		"Role", [Name],
		"Description", [Description]
	)
VAR _refTables =
	SELECTCOLUMNS(
		INFO.TABLES(),
			"Table ID", [ID],
			"Table", [Name]
	)

-- BASE

VAR _baseColumns =
	NATURALLEFTOUTERJOIN(
		SELECTCOLUMNS(
			INFO.VIEW.COLUMNS(),
			"Columns", BLANK(),
			[Table],
			"Column", [Name],
			[Description],
			"DAX", [Expression],
			"Format String", [FormatString],
			"Data Type", [DataType],
			"Is Hidden", [IsHidden],
			"Is Unique", [IsUnique],
			"Sorted By", [SortByColumn],
			"Category", [DataCategory]
		),
		SELECTCOLUMNS(
			COLUMNSTATISTICS(),
			"Table", [Table Name],
			"Column", [Column Name],
			"Lowest Value", CONVERT([Min], STRING),
			"Highest Value", CONVERT([Max], STRING),
			[Cardinality],
			"Value Length", [Max Length]
		)
	)
VAR _baseTables =
	VAR _baseTableRows =
		SELECTCOLUMNS(
			FILTER(_baseColumns, [Category] = "RowNumber"),
			[Table],
			"Rows", [Cardinality]
		)
	VAR _baseTableQueries =
		VAR _baseRefreshPolicyQueries =
			SELECTCOLUMNS(
				INFO.REFRESHPOLICIES(),
				"Table ID", [TableID],
				"Refresh Policy Query", [SourceExpression]
			)
		VAR _basePartitionQueries =
			SELECTCOLUMNS(
				GROUPBY(
					INFO.PARTITIONS(),
					[TableID],
					"Partition Query", MAXX(CURRENTGROUP(), [QueryDefinition])
				),
				"Table ID", [TableID],
				"Partition Query", [Partition Query]
			)
		RETURN
			SELECTCOLUMNS(
				NATURALLEFTOUTERJOIN(
					_basePartitionQueries,
					_baseRefreshPolicyQueries
				),
				"Table ID", [Table ID],
				"DAX or M-Query", COALESCE([Refresh Policy Query], [Partition Query])
			)
	RETURN
		NATURALLEFTOUTERJOIN(
			NATURALLEFTOUTERJOIN(
				SELECTCOLUMNS(
                    ADDCOLUMNS(
					    INFO.VIEW.TABLES(),
					"Is Calculated", IF([Expression] = BLANK(), "False", "True")
                    ),
					"Tables", BLANK(),
					"Table ID", [ID],
					"Table", [Name],
					[Description],
					"Is Hidden", [IsHidden],
					"Storage Mode", [StorageMode],
                    [Is Calculated]
				),
				_baseTableRows
			),
			_baseTableQueries
		)
VAR _baseDependencies =
	SELECTCOLUMNS(
		INFO.CALCDEPENDENCY(),
		"Object", [OBJECT],
		"Table", [TABLE],
		"Type", [OBJECT_TYPE],
		"Referenced Object", [REFERENCED_OBJECT],
		"Referenced Table", [REFERENCED_TABLE],
		"Referenced Type", [REFERENCED_OBJECT_TYPE]
	)
VAR _baseCalculationGroups =
	SELECTCOLUMNS(
		INFO.CALCULATIONGROUPS(),
		"Calculation Group ID", [ID],
		"Table ID", [TableID],
		"Table", MINX(FILTER(_refTables, [Table ID] = [TableID]), [Table]),
		[Description]
	)

-- FINAL

VAR _tables =
	SELECTCOLUMNS(
		_baseTables,
		[Tables], [Table], [Description], [Is Hidden], [Is Calculated], [Storage Mode], [Rows], [DAX or M-Query]
	)
VAR _columns =
	SELECTCOLUMNS(
		FILTER(
			_baseColumns,
			[Category] <> "RowNumber"
		),
		[Columns], [Table], [Column], [Description],[DAX],[Format String], [Data Type], [Is Hidden], [Is Unique], [Sorted By], [Lowest Value], [Highest Value], [Cardinality],
		"Value Length", COALESCE([Value Length], LEN([Highest Value]))
	)
VAR _measures =
	NATURALLEFTOUTERJOIN(
		NATURALLEFTOUTERJOIN(
			SELECTCOLUMNS(
				INFO.VIEW.MEASURES(),
				"Measures", BLANK(),
				"Measure", [Name],
				[Description],
				"Data Type", [DataType],
				"DAX", [Expression], 
				"Format String", [FormatStringDefinition], 
				[Table], 
				"Display Folder", [DisplayFolder],
				"Is Hidden", [IsHidden],
				[State],
				"DAX Length", LEN([Expression]),
				"Is Iterator",
					IF (
						COUNTROWS(
							FILTER(
								_refIteratorFunctions,
								CONTAINSSTRING(
									SUBSTITUTE(
									SUBSTITUTE(
									SUBSTITUTE(
									SUBSTITUTE(UPPER([Expression]),
										" ",""),
										UNICHAR(9), ""),
										UNICHAR(10), ""),
										UNICHAR(13), ""),
									[Value]
								)
							)
						) > 0,
						TRUE,
						FALSE
					),
				"Variables",
					VAR _exp =
						SUBSTITUTE(
						SUBSTITUTE(
						SUBSTITUTE(
						SUBSTITUTE(UPPER([Expression]),
							" ",""),
							"|", "~"),
							UNICHAR(9), ""),
							UNICHAR(10) & "VAR","|")
					VAR _cnt = PATHLENGTH(_exp) - 1 + IF(LEFT(_exp, 3) = "VAR", 1,0)
					RETURN
						IF(_cnt <= 0,BLANK(), _cnt)
			),
			SELECTCOLUMNS(
				GROUPBY(FILTER(_baseDependencies, [Referenced Type] = "MEASURE" && [Type] = "MEASURE"), [Object], "Referenced Measures", COUNTX(CURRENTGROUP(),1)),
					"Measure", [Object],
					[Referenced Measures]
			)
		),
		SELECTCOLUMNS(
		GROUPBY(FILTER(_baseDependencies, [Type] = "MEASURE"), [Object], "Referenced Objects", COUNTX(CURRENTGROUP(),1)),
			"Measure", [Object],
			[Referenced Objects]
		)
	)
VAR _relationships =
	SELECTCOLUMNS(
		INFO.VIEW.RELATIONSHIPS(),
		"Relationships", BLANK(),
		[Relationship],
		[Name],
		"Is Active", [IsActive],
		"Direction", [CrossFilteringBehavior],
		"From Table", [FromTable],
		"From Column", [FromColumn],
		"To Table", [ToTable],
		"To Column", [ToColumn],
		"Cardinality", [FromCardinality] & " to " & [ToCardinality]
	)
VAR _otherQueries =
	SELECTCOLUMNS(
		INFO.EXPRESSIONS(),
		"Other Queries", BLANK(),
		"Query Name", [Name],
		"Description", [Description],
		"M-Query", [Expression]
	)
VAR _incrementalTables =
	SELECTCOLUMNS(
		INFO.REFRESHPOLICIES(),
		"Incremental Tables", BLANK(),
		"Table", MINX(FILTER(_refTables, [Table ID] = [TableID]), [Table]),
		"Rolling Window History", [RollingWindowPeriods] & " " & MINX(FILTER(_refRefreshGranularity, [RollingWindowGranularity] = [Granularity ID]), [Granularity]),
		"Incremental Window", [IncrementalPeriods] & " " & MINX(FILTER(_refRefreshGranularity, [IncrementalGranularity] = [Granularity ID]), [Granularity]),
		"Incremental Window End",  [IncrementalPeriodsOffset] & " " & MINX(FILTER(_refRefreshGranularity, [IncrementalGranularity] = [Granularity ID]), [Granularity]) & " from Refresh Date",
		"Refresh Mode", MINX(FILTER(_refRefreshMode, [Refresh Mode ID] = [PolicyType]), [Refresh Mode])
	)
VAR _rlsRoles =
	SELECTCOLUMNS(_refRlsRoles,
		"RLS Roles", BLANK(),
		[Role],
		[Description]
	)
VAR _rlsFilters =
	VAR _baseRlsFilters =
		SELECTCOLUMNS(
			INFO.TABLEPERMISSIONS(),
			"Role ID", [RoleID],
			"RLS Table ID", [TableID],
			"DAX", [FilterExpression]
		)
	RETURN
		SELECTCOLUMNS(
			_baseRlsFilters,
			"RLS Filters", BLANK(),
			"Role", MINX(FILTER(_refRlsRoles, [Role ID] = [RLS Role ID]), [Role]),
			"Table", MINX(FILTER(_refTables, [Table ID] = [RLS Table ID]), [Table]),
			[DAX]
		)
VAR _calcGroups =
	SELECTCOLUMNS(
		_baseCalculationGroups,
		"Calculation Groups", BLANK(),
		"Calculation Group", [Table],
		[Description]
	)
VAR _calcItems =
	SELECTCOLUMNS(
		INFO.CALCULATIONITEMS(),
		"Calculation Items", BLANK(),
		"Calculation Group", MINX(FILTER(_baseCalculationGroups, [Calculation Group ID] = [CalculationGroupID]), [Table]),
		"Calculation Item", [Name],
		[Description],
		"DAX", [Expression]
	)
VAR _dependencies =
	SELECTCOLUMNS(
		_baseDependencies,
		"Dependencies", BLANK(),
		[Object], [Table], [Type], [Referenced Object], [Referenced Table], [Referenced Type]
	)
-- VAR _functions =
-- 	SELECTCOLUMNS(
-- 		INFO.USERDEFINEDFUNCTIONS(),
-- 		"User Functions", BLANK(),
-- 		[Name],
-- 		[Description],
-- 		"DAX", [Expression],
-- 		"Is Hidden", [IsHidden]
-- 	)

-- OUTPUT

EVALUATE _info
EVALUATE _tables
EVALUATE _columns
EVALUATE _measures
EVALUATE _relationships
EVALUATE _otherQueries
EVALUATE _incrementalTables
EVALUATE _rlsRoles
EVALUATE _rlsFilters
EVALUATE _calcGroups
EVALUATE _calcItems
EVALUATE _dependencies
-- EVALUATE _functions
